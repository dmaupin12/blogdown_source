library(leaflet.extras)
## Loading required package: leaflet
library(openxlsx)
## Warning: package 'openxlsx' was built under R version 3.4.3
#cxo_leases<- readOGR(".","cxo_leases_2018-01-22", verbose=F)
cdev_leases<- readOGR(".","Cen_One_02_26_18_MA_Shapefile_63635", verbose=F)
cdev_leases2<- readOGR(".","Und_Cen_02_26_18_MA_Shapefile_63636", verbose=F)
cdev_legacy<- readOGR(".","cdev_legacy", verbose=F)
cdev_wells<- read.xlsx("cdev.xlsx",sheet=1)
map1<-leaflet(width='100%',height = 550) %>%
  # Base groups
  addTiles(group = "Esri.WorldTopoMap")%>%
  addProviderTiles(providers$Esri.WorldTopoMap)%>%
  addPolygons(data = cdev_leases, fillOpacity = .5,fillColor = 'blue',stroke=F,group = "Acquisition")%>%
  addPolygons(data = cdev_leases2, fillOpacity = .5,fillColor = 'red',stroke=F,group = "Divestiture")%>%
  addPolygons(data = cdev_legacy, fillOpacity = .5,fillColor = 'grey',stroke=F,group = "Legacy")%>%
  addCircleMarkers(data=cdev_wells, ~Longitude, ~Latitude,radius=5,stroke = F,color=~c('orange','black','green','purple','gold','dark blue'),fillOpacity=1, group = "Well Highlights", 
                   popup=paste("Well Name:", cdev_wells$Well.Name,"<br>",
                               "IP/30 Bo/d:",cdev_wells$`IP30.Bo/d`,"<br>",
                               "Formation", cdev_wells$Formation))
 setView(map1,lng=-103.57, lat=31.47, zoom=8)%>%
  # Overlay groups
  addLayersControl(
    overlayGroups = c( "Acquisition",'Divestiture','Legacy','Well Highlights'),
    options = layersControlOptions(collapsed = FALSE),position='topleft')%>%
  addFullscreenControl()
